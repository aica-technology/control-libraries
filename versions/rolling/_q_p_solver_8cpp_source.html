35bool QPSolver::init_solver() {
36 // Clear the solver
37 this->solver_.data()->clearHessianMatrix();
38 this->solver_.data()->clearLinearConstraintsMatrix();
39 this->solver_.clearSolver();
40
41 // Initialize the matrices
42 this->hessian_ = Eigen::SparseMatrix<double>(this->nb_joints_ + 1, this->nb_joints_ + 1);
43 this->gradient_ = Eigen::VectorXd::Zero(this->nb_joints_ + 1);
44 this->constraint_matrix_ = Eigen::SparseMatrix<double>(3 * this->nb_joints_ + 1 + 2, this->nb_joints_ + 1);
45 this->lower_bound_constraints_ = Eigen::VectorXd::Zero(3 * this->nb_joints_ + 1 + 2);
46 this->upper_bound_constraints_ = Eigen::VectorXd::Zero(3 * this->nb_joints_ + 1 + 2);
47
48 // reserve the size of the matrices
49 this->hessian_.reserve(this->nb_joints_ * this->nb_joints_ + 1);
50 this->constraint_matrix_.reserve(
51 5 * this->nb_joints_ + 2 * (this->nb_joints_ * this->nb_joints_ + this->nb_joints_) + 4 * this->nb_joints_ + 3);
52
53 // configure the QP problem
54 this->solver_.settings()->setVerbosity(false);
55 this->solver_.settings()->setWarmStart(true);
56
57 // joint dependent constraints
58 for (unsigned int n = 0; n < this->nb_joints_; ++n) {
59 // joint limits
60 this->constraint_matrix_.coeffRef(n, n) = 1.0;
61
62 // joint velocity limits
63 this->constraint_matrix_.coeffRef(n + this->nb_joints_, n) = 1.0;
64 this->constraint_matrix_.coeffRef(n + this->nb_joints_, this->nb_joints_) = this->velocity_limit_(n);
65 this->upper_bound_constraints_(n + this->nb_joints_) = std::numeric_limits<double>::infinity();
66 this->constraint_matrix_.coeffRef(n + 2 * this->nb_joints_, n) = 1.0;
67 this->constraint_matrix_.coeffRef(n + 2 * this->nb_joints_, this->nb_joints_) = -this->velocity_limit_(n);
68 this->lower_bound_constraints_(n + 2 * this->nb_joints_) = -std::numeric_limits<double>::infinity();
69 }
70
71 // time constraint
72 this->constraint_matrix_.coeffRef(3 * this->nb_joints_, this->nb_joints_) = 1.0;
73 this->upper_bound_constraints_(3 * this->nb_joints_) = std::numeric_limits<double>::infinity();
74
75 // cartesian velocity constraints
76 this->upper_bound_constraints_(3 * this->nb_joints_ + 1) = std::numeric_limits<double>::infinity();
77 this->upper_bound_constraints_(3 * this->nb_joints_ + 2) = std::numeric_limits<double>::infinity();
78
79 // set the initial data of the QP solver_
80 this->solver_.data()->setNumberOfVariables(static_cast<int>(this->nb_joints_) + 1);
81 this->solver_.data()->setNumberOfConstraints(this->lower_bound_constraints_.size());
82
83 if (!this->solver_.data()->setHessianMatrix(this->hessian_)) {
84 return false;
85 }
86 if (!this->solver_.data()->setGradient(this->gradient_)) {
87 return false;
88 }
89 if (!this->solver_.data()->setLinearConstraintsMatrix(this->constraint_matrix_)) {
90 return false;
91 }
92 if (!this->solver_.data()->setLowerBound(this->lower_bound_constraints_)) {
93 return false;
94 }
95 if (!this->solver_.data()->setUpperBound(this->upper_bound_constraints_)) {
96 return false;
97 }
98
99 // instantiate the solver_
100 return this->solver_.initSolver();
101}

117void QPSolver::set_matrices(
118 std::vector<Eigen::Triplet<double>> coefficients, const QPInverseVelocityParameters& parameters,
119 const state_representation::JointPositions& joint_positions,
120 const state_representation::CartesianPose& full_displacement, const Eigen::VectorXd& delta_robot,
121 const Eigen::MatrixXd& jacobian) {
122 using namespace std::chrono;
123
124 // update the hessian matrix
125 this->hessian_.setFromTriplets(coefficients.begin(), coefficients.end());
126
127 // update the gradient vector
128 this->gradient_.head(this->nb_joints_) = -parameters.proportional_gain * delta_robot.transpose() * jacobian;
129
130 // lower bound constraints
131 this->lower_bound_constraints_(3 * this->nb_joints_) = duration_cast<duration<float>>(parameters.dt).count();
132
133 for (unsigned int n = 0; n < this->nb_joints_; ++n) {
134 this->lower_bound_constraints_(n) = this->lower_position_limit_(n) - joint_positions.data()(n);
135 this->upper_bound_constraints_(n) = this->upper_position_limit_(n) - joint_positions.data()(n);
136 }
137
138 // update the constraint matrix
139 this->constraint_matrix_.coeffRef(3 * this->nb_joints_ + 1, this->nb_joints_) = parameters.linear_velocity_limit;
140 this->constraint_matrix_.coeffRef(3 * this->nb_joints_ + 2, this->nb_joints_) = parameters.angular_velocity_limit;
141 this->lower_bound_constraints_(3 * this->nb_joints_ + 1) = full_displacement.get_position().norm();
142 this->lower_bound_constraints_(3 * this->nb_joints_ + 2) = full_displacement.get_orientation().vec().norm();
143}
