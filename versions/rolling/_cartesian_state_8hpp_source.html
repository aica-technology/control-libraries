609inline state_representation::CartesianStateVariable string_to_cartesian_state_variable(const std::string& variable) {
610 std::string case_insensitive_variable;
611 std::transform(variable.begin(), variable.end(), std::back_inserter(case_insensitive_variable), [](unsigned char c) {
612 return std::tolower(c);
613 });
614 if (case_insensitive_variable == "position") {
615 return CartesianStateVariable::POSITION;
616 } else if (case_insensitive_variable == "orientation") {
617 return CartesianStateVariable::ORIENTATION;
618 } else if (case_insensitive_variable == "pose") {
619 return CartesianStateVariable::POSE;
620 } else if (case_insensitive_variable == "linear_velocity") {
621 return CartesianStateVariable::LINEAR_VELOCITY;
622 } else if (case_insensitive_variable == "angular_velocity") {
623 return CartesianStateVariable::ANGULAR_VELOCITY;
624 } else if (case_insensitive_variable == "twist") {
625 return CartesianStateVariable::TWIST;
626 } else if (case_insensitive_variable == "linear_acceleration") {
627 return CartesianStateVariable::LINEAR_ACCELERATION;
628 } else if (case_insensitive_variable == "angular_acceleration") {
629 return CartesianStateVariable::ANGULAR_ACCELERATION;
630 } else if (case_insensitive_variable == "acceleration") {
631 return CartesianStateVariable::ACCELERATION;
632 } else if (case_insensitive_variable == "force") {
633 return CartesianStateVariable::FORCE;
634 } else if (case_insensitive_variable == "torque") {
635 return CartesianStateVariable::TORQUE;
636 } else if (case_insensitive_variable == "twist") {
637 return CartesianStateVariable::TWIST;
638 } else if (case_insensitive_variable == "wrench") {
639 return CartesianStateVariable::WRENCH;
640 } else if (case_insensitive_variable == "all") {
641 return CartesianStateVariable::ALL;
642 } else {
643 throw exceptions::InvalidStateVariableException("Invalid Cartesian state variable: " + variable);
644 }
645 __builtin_unreachable();
646}
