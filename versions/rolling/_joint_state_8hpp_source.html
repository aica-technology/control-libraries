590inline state_representation::JointStateVariable string_to_joint_state_variable(const std::string& variable) {
591 std::string case_insensitive_variable;
592 std::transform(variable.begin(), variable.end(), std::back_inserter(case_insensitive_variable), [](unsigned char c) {
593 return std::tolower(c);
594 });
595 if (case_insensitive_variable == "positions") {
596 return JointStateVariable::POSITIONS;
597 } else if (case_insensitive_variable == "velocities") {
598 return JointStateVariable::VELOCITIES;
599 } else if (case_insensitive_variable == "accelerations") {
600 return JointStateVariable::ACCELERATIONS;
601 } else if (case_insensitive_variable == "torques") {
602 return JointStateVariable::TORQUES;
603 } else if (case_insensitive_variable == "all") {
604 return JointStateVariable::ALL;
605 } else {
606 throw exceptions::InvalidStateVariableException("Invalid joint state variable: " + variable);
607 }
608 __builtin_unreachable();
609}
