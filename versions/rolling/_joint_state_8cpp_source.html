204void JointState::set_state_variable(const Eigen::VectorXd& new_value, const JointStateVariable& state_variable_type) {
205 auto expected_size = get_state_variable_size_factor(state_variable_type) * this->get_size();
206 if (new_value.size() != expected_size) {
207 throw exceptions::IncompatibleSizeException(
208 "Input is of incorrect size, expected " + std::to_string(expected_size) + ", got "
209 + std::to_string(new_value.size())
210 );
211 }
212 switch (state_variable_type) {
213 case JointStateVariable::POSITIONS:
214 this->positions_ = new_value;
215 break;
216 case JointStateVariable::VELOCITIES:
217 this->velocities_ = new_value;
218 break;
219 case JointStateVariable::ACCELERATIONS:
220 this->accelerations_ = new_value;
221 break;
222 case JointStateVariable::TORQUES:
223 this->torques_ = new_value;
224 break;
225 case JointStateVariable::ALL: {
226 auto size = this->get_size();
227 this->positions_ = new_value.head(size);
228 this->velocities_ = new_value.segment(size, size);
229 this->accelerations_ = new_value.segment(2 * size, size);
230 this->torques_ = new_value.tail(size);
231 break;
232 }
233 }
234 this->set_empty(false);
235}

359void JointState::clamp_state_variable(
360 const Eigen::ArrayXd& max_absolute_value_array, const JointStateVariable& state_variable_type,
361 const Eigen::ArrayXd& noise_ratio_array
362) {
363 Eigen::VectorXd state_variable = this->get_state_variable(state_variable_type);
364 long expected_size = state_variable.size();
365 if (max_absolute_value_array.size() != expected_size) {
366 throw IncompatibleSizeException(
367 "Array of max values is of incorrect size: expected " + std::to_string(expected_size) + ", given "
368 + std::to_string(max_absolute_value_array.size())
369 );
370 }
371
372 if (noise_ratio_array.size() != expected_size) {
373 throw IncompatibleSizeException(
374 "Array of max values is of incorrect size: expected " + std::to_string(expected_size) + ", given "
375 + std::to_string(noise_ratio_array.size())
376 );
377 }
378 for (int i = 0; i < expected_size; ++i) {
379 if (noise_ratio_array(i) != 0.0 && abs(state_variable(i)) < noise_ratio_array(i) * max_absolute_value_array(i)) {
380 // apply dead zone
381 state_variable(i) = 0.0;
382 } else if (abs(state_variable(i)) > max_absolute_value_array(i)) {
383 // clamp to max value
384 state_variable(i) *= max_absolute_value_array(i) / abs(state_variable(i));
385 }
386 }
387 this->set_state_variable(state_variable, state_variable_type);
388}

406double JointState::dist(const JointState& state, const JointStateVariable& state_variable_type) const {
407 if (this->is_incompatible(state)) {
408 throw IncompatibleStatesException("The two joint states are incompatible, check name, joint names and order or size"
409 );
410 }
411 // calculation
412 double result = 0;
413 if (state_variable_type == JointStateVariable::POSITIONS || state_variable_type == JointStateVariable::ALL) {
414 result += (this->get_positions() - state.get_positions()).norm();
415 }
416 if (state_variable_type == JointStateVariable::VELOCITIES || state_variable_type == JointStateVariable::ALL) {
417 result += (this->get_velocities() - state.get_velocities()).norm();
418 }
419 if (state_variable_type == JointStateVariable::ACCELERATIONS || state_variable_type == JointStateVariable::ALL) {
420 result += (this->get_accelerations() - state.get_accelerations()).norm();
421 }
422 if (state_variable_type == JointStateVariable::TORQUES || state_variable_type == JointStateVariable::ALL) {
423 result += (this->get_torques() - state.get_torques()).norm();
424 }
425 return result;
426}

437bool JointState::is_incompatible(const State& state) const {
438 try {
439 auto other = dynamic_cast<const JointState&>(state);
440 if (this->names_.size() != other.names_.size()) {
441 return true;
442 }
443 for (unsigned int i = 0; i < this->names_.size(); ++i) {
444 if (this->names_[i] != other.names_[i]) {
445 return true;
446 }
447 }
448 return false;
449 } catch (const std::bad_cast& ex) {
450 throw exceptions::InvalidCastException(
451 std::string("Could not cast the given object to a JointState: ") + ex.what()
452 );
453 }

471void JointState::multiply_state_variable(const Eigen::MatrixXd& lambda, const JointStateVariable& state_variable_type) {
472 Eigen::VectorXd state_variable = this->get_state_variable(state_variable_type);
473 long expected_size = state_variable.size();
474 if (lambda.rows() != expected_size || lambda.cols() != expected_size) {
475 throw IncompatibleSizeException(
476 "Gain matrix is of incorrect size: expected " + std::to_string(expected_size) + "x"

478 + std::to_string(lambda.cols())
479 );
480 }
481 this->set_state_variable(lambda * this->get_state_variable(state_variable_type), state_variable_type);
482}
