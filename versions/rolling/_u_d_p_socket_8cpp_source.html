23void UDPSocket::open_socket(bool bind_socket) {
24 try {
25 this->addr_len_ = sizeof(this->server_address_);
26 this->server_address_.sin_family = AF_INET;
27 this->server_address_.sin_addr.s_addr = inet_addr(this->config_.ip_address.c_str());
28 this->server_address_.sin_port = htons(this->config_.port);
29 } catch (const std::exception& ex) {
30 throw exceptions::SocketConfigurationException("Socket configuration failed: " + std::string(ex.what()));
31 }
32
33 this->server_fd_ = socket(AF_INET, SOCK_DGRAM, IPPROTO_UDP);
34 if (this->server_fd_ < 0) {
35 throw exceptions::SocketConfigurationException("Opening socket failed");
36 }
37 if (this->config_.enable_reuse) {
38 const int opt_reuse = 1;
39 if (setsockopt(this->server_fd_, SOL_SOCKET, SO_REUSEADDR | SO_REUSEPORT, &opt_reuse, sizeof(opt_reuse)) != 0) {
40 throw exceptions::SocketConfigurationException("Setting socket option (enable reuse) failed");
41 }
42 }
43 if (bind_socket) {
44 if (bind(this->server_fd_, (sockaddr*) &(this->server_address_), sizeof(server_address_)) != 0) {
45 throw exceptions::SocketConfigurationException("Binding socket failed.");
46 }
47 }
48
49 if (this->config_.timeout_duration_sec > 0.0
50 && this->config_.timeout_duration_sec < std::numeric_limits<double>::max()) {
51 timeval timeout{};
52 auto secs = std::floor(this->config_.timeout_duration_sec);
53 timeout.tv_sec = static_cast<long>(secs);
54 timeout.tv_usec = static_cast<long>((this->config_.timeout_duration_sec - secs) * 1e6);
55 if (setsockopt(this->server_fd_, SOL_SOCKET, SO_RCVTIMEO, &timeout, sizeof(timeout)) != 0) {
56 throw exceptions::SocketConfigurationException("Setting socket timeout failed.");
57 }
58 }
59}

61bool UDPSocket::recvfrom(sockaddr_in& address, std::string& buffer) {
62 std::vector<char> local_buffer(this->config_.buffer_size);
63 auto receive_length = ::recvfrom(
64 this->server_fd_, local_buffer.data(), this->config_.buffer_size, 0, (sockaddr*) &(address), &(this->addr_len_));
65 if (receive_length < 0) {
66 return false;
67 }
68 buffer.assign(local_buffer.data(), local_buffer.size());
69 return true;
