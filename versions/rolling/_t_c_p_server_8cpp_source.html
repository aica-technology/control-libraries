18void TCPServer::on_open() {
19 try {
20 bzero((char*) &this->server_address_, sizeof(this->server_address_));
21 this->server_address_.sin_family = AF_INET;
22 this->server_address_.sin_addr.s_addr = htonl(INADDR_ANY);
23 this->server_address_.sin_port = htons(this->config_.port);
24 } catch (const std::exception& ex) {
25 throw exceptions::SocketConfigurationException("Socket configuration failed: " + std::string(ex.what()));
26 }
27
28 // open stream oriented socket with internet address
29 this->server_fd_ = socket(AF_INET, SOCK_STREAM, 0);
30 if (this->server_fd_ < 0) {
31 throw exceptions::SocketConfigurationException("Opening socket failed");
32 }
33 if (this->config_.enable_reuse) {
34 const int opt_reuse = 1;
35 if (setsockopt(this->server_fd_, SOL_SOCKET, SO_REUSEADDR | SO_REUSEPORT, &opt_reuse, sizeof(opt_reuse)) != 0) {
36 throw exceptions::SocketConfigurationException("Setting socket option (enable reuse) failed");
37 }
38 }
39 if (bind(this->server_fd_, (sockaddr*) &(this->server_address_), sizeof(this->server_address_)) != 0) {
40 throw exceptions::SocketConfigurationException("Binding socket failed");
41 }
42 // listen for only 1 request at a time
43 listen(this->server_fd_, 1);
44 // receive a request from client using accept
45 sockaddr_in new_socket_address{};
46 socklen_t new_addr_len = sizeof(new_socket_address);
47 // accept, create a new socket descriptor to handle the new connection with client
48 this->socket_fd_ = accept(this->server_fd_, (sockaddr*) &new_socket_address, &new_addr_len);
49 if (this->socket_fd_ < 0) {
50 throw exceptions::SocketConfigurationException("Connecting server failed");
51 }
52}
