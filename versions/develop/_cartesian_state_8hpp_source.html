592inline Eigen::VectorXd CartesianState::get_state_variable(const CartesianStateVariable& state_variable_type) const {
593 switch (state_variable_type) {
594 case CartesianStateVariable::POSITION:
595 return this->get_position();
596
597 case CartesianStateVariable::ORIENTATION:
598 return this->get_orientation_coefficients();
599
600 case CartesianStateVariable::POSE:
601 return this->get_pose();
602
603 case CartesianStateVariable::LINEAR_VELOCITY:
604 return this->get_linear_velocity();
605
606 case CartesianStateVariable::ANGULAR_VELOCITY:
607 return this->get_angular_velocity();
608
609 case CartesianStateVariable::TWIST:
610 return this->get_twist();
611
612 case CartesianStateVariable::LINEAR_ACCELERATION:
613 return this->get_linear_acceleration();
614
615 case CartesianStateVariable::ANGULAR_ACCELERATION:
616 return this->get_angular_acceleration();
617
618 case CartesianStateVariable::ACCELERATION:
619 return this->get_acceleration();
620
621 case CartesianStateVariable::FORCE:
622 return this->get_force();
623
624 case CartesianStateVariable::TORQUE:
625 return this->get_torque();
626
627 case CartesianStateVariable::WRENCH:
628 return this->get_wrench();
629
630 case CartesianStateVariable::ALL:
631 Eigen::VectorXd all_fields(25);
632 all_fields << this->get_pose(), this->get_twist(), this->get_acceleration(), this->get_wrench();
633 return all_fields;
634 }
635 // this never goes here but is compulsory to avoid a warning
636 return Eigen::Vector3d::Zero();
637}
