650inline void JointState::set_all_state_variables(const Eigen::VectorXd& new_values) {
651 if (new_values.size() != 4 * this->get_size()) {
652 throw state_representation::exceptions::IncompatibleSizeException(
653 "Input is of incorrect size: expected " + std::to_string(this->get_size()) + ", given "
654 + std::to_string(new_values.size()));
655 }
656 this->set_positions(new_values.segment(0, this->get_size()));
657 this->set_velocities(new_values.segment(this->get_size(), this->get_size()));
658 this->set_accelerations(new_values.segment(2 * this->get_size(), this->get_size()));
659 this->set_torques(new_values.segment(3 * this->get_size(), this->get_size()));
660}
