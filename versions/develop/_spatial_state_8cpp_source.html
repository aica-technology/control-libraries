34bool SpatialState::is_incompatible(const State& state) const {
35 try {
36 auto other = dynamic_cast<const SpatialState&>(state);
37 // the three conditions for compatibility are:
38 // 1) this name matches other reference frame (this is parent transform of other)
39 // 2) this reference frame matches other name (this is child transform of other)
40 // 3) this reference frame matches other reference frame (this is sibling transform of other)
41 return (this->get_name() != other.reference_frame_) && (this->reference_frame_ != other.get_name())
42 && (this->get_reference_frame() != other.reference_frame_);
43 } catch (const std::bad_cast& ex) {
44 throw exceptions::InvalidCastException(
45 std::string("Could not cast the given object to a SpatialState: ") + ex.what());
46 }
47}
