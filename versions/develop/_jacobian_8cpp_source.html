186bool Jacobian::is_incompatible(const State& state) const {
187 try {
188 switch (state.get_type()) {
189 case StateType::JACOBIAN: {
190 auto other = dynamic_cast<const Jacobian&>(state);
191 if (this->cols_ != other.joint_names_.size()) {
192 return true;
193 }
194 if (this->reference_frame_ != other.reference_frame_) {
195 return true;
196 }
197 for (unsigned int i = 0; i < this->cols_; ++i) {
198 if (this->joint_names_[i] != other.joint_names_[i]) {
199 return true;
200 }
201 }
202 return false;
203 }
204 case StateType::JOINT_STATE:
205 case StateType::JOINT_POSITIONS:
206 case StateType::JOINT_VELOCITIES:
207 case StateType::JOINT_ACCELERATIONS:
208 case StateType::JOINT_TORQUES: {
209 auto other = dynamic_cast<const JointState&>(state);
210 if (this->cols_ != other.get_names().size()) {
211 return true;
212 }
213 for (unsigned int i = 0; i < this->cols_; ++i) {
214 if (this->joint_names_[i] != other.get_names()[i]) {
215 return true;
216 }
217 }
218 return false;
219 }
220 case StateType::CARTESIAN_STATE:
221 case StateType::CARTESIAN_POSE:
222 case StateType::CARTESIAN_TWIST:
223 case StateType::CARTESIAN_ACCELERATION:
224 case StateType::CARTESIAN_WRENCH: {
225 auto other = dynamic_cast<const CartesianState&>(state);
226 if (this->reference_frame_ != other.get_reference_frame()) {
227 return true;
228 }
229 return false;
230 }
231 default:
232 return true;
233 }
234 } catch (const std::bad_cast& ex) {
235 throw exceptions::InvalidCastException(std::string("Could not cast the given object: ") + ex.what());
236 }
237}

359Jacobian operator*(const CartesianPose& pose, const Jacobian& jacobian) {
360 // check compatibility
361 if (jacobian.is_empty()) {
362 throw EmptyStateException(jacobian.get_name() + " state is empty");
363 }
364 if (pose.is_empty()) {
365 throw EmptyStateException(pose.get_name() + " state is empty");
366 }
367 if (pose.get_name() != jacobian.get_reference_frame()) {
368 throw IncompatibleStatesException("The Jacobian and the input CartesianPose are incompatible, expected pose of "
369 + jacobian.get_reference_frame() + " got " + pose.get_name());
370 }
371 // number of rows of the jacobian should be 6 (incorrect if it has been transposed before)
372 // FIXME transpose is weird and confusing with the statement above (also what does it mean for the incompatibility?)
373 if (jacobian.rows_ != 6) {
374 throw IncompatibleStatesException(
375 "The Jacobian and the input CartesianPose are incompatible, the Jacobian has probably been transposed before");
376 }
377 Jacobian result(jacobian);
378 // change the reference frame of all the columns
379 for (unsigned int i = 0; i < jacobian.cols_; ++i) {
380 // update position part
381 result.data_.col(i).head(3) = pose.get_orientation() * jacobian.data_.col(i).head(3);
382 // update orientation part
383 result.data_.col(i).tail(3) = pose.get_orientation() * jacobian.data_.col(i).tail(3);
384 }
385 // change the reference frame
386 result.reference_frame_ = pose.get_reference_frame();
387 return result;
388}
